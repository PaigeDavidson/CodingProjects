Description

This is an Image carosel that can display any number of images.

You can add your images by adding another image path inside the images div in the HTML or replacing the src of the existing images with your own.

It is important to remember to link your own CSS and JS files at the top of your HTML file in order for the component to work properly.

JS

        
const imagesContainer = document.querySelector('.images');
let images = document.querySelectorAll('.image');
let index = 0;

function showImage(index) {
    const translateValue = -index * 100 + '%';
    imagesContainer.style.transition = "transform 0.5s ease-in-out";
    imagesContainer.style.transform = `translateX(${translateValue})`;
}

function nextImage() {
    // Clone and append the images
    let clone = images[index].cloneNode(true);
    imagesContainer.appendChild(clone);

    // Update the images NodeList
    images = document.querySelectorAll('.image');

    // Update the index and show the next image
    index++;
    showImage(index);
}

// Set an interval to transition to the next image every 3 seconds
setInterval(nextImage, 3000);
